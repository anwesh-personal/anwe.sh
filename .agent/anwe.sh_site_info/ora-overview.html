SDK

Oraya SDK

One interface to rule them all. Implement this and ORA can control your system.

oraya-connector.ts 
interface OrayaConnector {
    // What actions are available?
    getSchema(): {
        name: string;
        version: string;
        actions: Record<string, ActionDef>;
    };
    
    // Execute an action
    executeAction(
        action: string,
        data: any
    ): Promise<any>;
}

// Example: Website connector
const website = createOrayaHandler({
    secret: process.env.ORA_SECRET,
    actions: {
        'posts.publish': async ({ id }) => {
            return await publishPost(id);
        },
        'memory.store': async ({ content, type }) => {
            return await storeMemory(content, type);
        }
    }
});
